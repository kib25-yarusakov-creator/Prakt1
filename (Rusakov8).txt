#include <stdio.h>
#include <string.h>

unsigned long long fact(int n) {
    unsigned long long f = 1;
    for (int i = 2; i <= n; i++)
        f *= i;
    return f;
}

int main() {
    char word[15];
    int count[256] = {0};

    printf("Слово: ");
    scanf("%14s", word);

    int len = strlen(word);

    for (int i = 0; i < len; i++)
        count[(unsigned char)word[i]]++;

    unsigned long long result = fact(len);

    for (int i = 0; i < 256; i++) {
        if (count[i] > 1)
            result /= fact(count[i]);
    }

    printf("Кількість анаграм: %llu\n", result);

    return 0;
}
