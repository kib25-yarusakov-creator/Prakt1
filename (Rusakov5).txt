#include <stdio.h>

int main() {
    int n;
    const int MOD = 12345;

    printf("Введіть довжину: ");
    scanf("%d", &n);

    int a = 1, b = 1, c = 0;

    for (int i = 2; i <= n; i++) {
        int na = (a + b + c) % MOD;
        int nb = a;
        int nc = b;

        a = na;
        b = nb;
        c = nc;
    }

    printf("Кількість шуканих: %d", (a + b + c) % MOD);

    return 0;
}
