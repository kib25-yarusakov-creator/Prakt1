#include <stdio.h>

int gcd(int a, int b) {
    while (b) {
        int t = a % b;
        a = b;
        b = t;
    }
    return a;
}

int main() {
    int p, x, ans;
    scanf("%d", &p);

    scanf("%d", &ans);
    while (--p) {
        scanf("%d", &x);
        ans = ans / gcd(ans, x) * x;
    }

    printf("%d", ans);

    return 0;
}
