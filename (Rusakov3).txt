#include <stdio.h>

int main() {
    int p;
    scanf("%d", &p);

    long long a = 1, b = 1, c = 1, d = 1;

    for (int i = 2; i <= p; i++) {
        long long na = c + d;
        long long nb = a;
        long long nc = a + b;
        long long nd = c;

        a = na;
        b = nb;
        c = nc;
        d = nd;
    }

    if (p == 1)
        printf("2");
    else
        printf("%lld", a + b + c + d);

    return 0;
}
